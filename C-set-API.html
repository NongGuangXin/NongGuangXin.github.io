C++ set

template< class Key, class Compare = std::less<Key>, class Allocator = std::allocator<Key>>
class set;

namespace pmr {
    template <class Key, class Compare = std::less<Key>>
    using set = std::set<Key, Compare, std::pmr::polymorphic_allocator<Key>>;
}

std::set 是关联容器，含有 Key 类型对象的已排序集。用比较函数 比较 (Compare) 进行排序。搜索、移除和插入拥有对数复杂度。 set 通常以红黑树实现。

在每个标准库使用比较 (Compare) 概念的场所，用等价关系确定唯一性。不精确地说，若二个对象 a 与 b 相互间既不比较大于亦不比较小于： !comp(a, b) && !comp(b, a) ，则认为它们等价。

成员类型

成员类型 定义
key_type Key
value_type Key
size_type 无符号整数类型（通常是 std::size_t ）
difference_type 有符号整数类型（通常是 std::ptrdiff_t ）
key_compare Compare
value_compare Compare
allocator_type Allocator
reference value_type&
const_reference const value_type&
pointer std::allocator_traits<Allocator>::pointer
const_pointer std::allocator_traits<Allocator>::const_pointer
iterator 遗留双向迭代器 (LegacyRandomAccessIterator)
const_iterator 常双向迭代器
reverse_iterator std::reverse_iterator<iterator>
const_reverse_iterator std::reverse_iterator<const_iterator>

构造

构造函数

  1. 创建一个空set.

    set();, explicit set( const Allocator& alloc );
    explicit set( const Compare& comp, const Allocator& alloc = Allocator() );

    std::set<std::string> v();

  2. 初始化列表

    set( std::initializer_list<value_type> init, const Compare& comp = Compare(), const Allocator& alloc = Allocator() );

    std::set<std::string> v{"aaa", "bbb", "ccc"};
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; });
// aaa bbb ccc

  3. 以一对迭代器创建

    template< class InputIt > set( InputIt first, InputIt last, const Compare& comp = Compare(), const Allocator& alloc = Allocator() );

    std::vector<std::string> v1{"aaa", "bbb", "ccc"};
std::set<std::string> v(v1.begin(), v1.end());
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; });
// aaa bbb ccc

  4. 以其他的vector容器创建

    set( const set& other ); ,set( const set& other, const Allocator& alloc );

    std::set<std::string> v1{"aaa", "bbb", "ccc"};
std::set<std::string> v(v1);
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; });
// aaa bbb ccc

    还可以使用移动语义

    set( set&& other );, set( set&& other, const Allocator& alloc );

    std::set<std::string> v1{"aaa", "bbb", "ccc"};
std::set<std::string> v(std::move(v1));
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; });
// aaa bbb ccc

operator=

set& operator=( const set& other );
set& operator=( set&& other ) noexcept
set& operator=( std::initializer_list<value_type> ilist );

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::set<std::string> v1 = v;
std::set<std::string> v2 = std::move(v);
std::set<std::string> v3 = {"aaa", "bbb", "ccc"};
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; });
// aaa bbb ccc

get_allocator

返回与容器关联的分配器。allocator_type get_allocator() const noexcept;

std::set<std::string> v;
std::set<std::string>::allocator_type alloc_type = v.get_allocator();

迭代器

begin - end

iterator begin() noexcept; const_iterator begin() const noexcept;
iterator end() noexcept; const_iterator end() const noexcept;

返回指向容器开头和结尾的迭代器。

std::set<std::string> v{"aaa", "bbb", "ccc"};
for (std::set<std::string>::iterator i = v.begin(); i != v.end(); i++)
    std::cout << *i << " ";// aaa bbb ccc

cbegin - cend

const_iterator cbegin() const noexcept; const_iterator cend() const noexcept;

返回指向容器开头和结尾的常量迭代器。

std::set<std::string> v{"aaa", "bbb", "ccc"};
for (std::set<std::string>::const_iterator i = v.begin(); i != v.end(); i++)
    std::cout << *i << " ";// aaa bbb ccc

rbegin - rend

reverse_iterator rbegin() noexcept; const_reverse_iterator rbegin() const noexcept;
const_reverse_iterator rend() const noexcept;

返回指向容器开头和结尾的反向迭代器。

std::set<std::string> v{"aaa", "bbb", "ccc"};
for (std::set<std::string>::reverse_iterator i = v.rbegin(); i != v.rend(); i++)
    std::cout << *i << " ";// ccc bbb aaa

crbegin - crend

const_reverse_iterator crbegin() const noexcept;
const_reverse_iterator crend() const noexcept;

返回指向容器开头和结尾的反向常量迭代器。

std::set<std::string> v{"aaa", "bbb", "ccc"};
for (std::set<std::string>::reverse_iterator i = v.rbegin(); i != v.rend(); i++)
    std::cout << *i << " ";// ccc bbb aaa

容量

empty

[[nodiscard]] bool empty() const noexcept;

判断容器是否为空，空则为true

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::cout << std::boolalpha << v.empty() << std::endl; // false

size

size_type size() const noexcept;

返回容器中的元素数, 即 std::distance(begin(), end()) 。

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::cout << v.size() << std::endl; // 3

max_size

size_type max_size() const noexcept;

返回根据系统或库实现限制的容器可保有的元素最大数量

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::cout << v.max_size() << std::endl; // 144115188075855871

修改器

clear

void clear() noexcept;

擦除所有元素。此调用后 size() 返回零。非法化任何指代所含元素的引用、指针或迭代器。任何尾后迭代器亦被非法化

std::set<std::string> v{"aaa", "bbb", "ccc"};v.clear();
std::cout << v.size() << std::endl; // 0

insert

插入元素到容器中的指定位置。

  1. 插入 value返回由指向被插入元素（或阻止插入的元素）的迭代器和若插入发生则设为 true 的 bool 值。 支持移动语义。

std::pair<iterator,bool> insert( const value_type& value );
std::pair<iterator,bool> insert( value_type&& value );

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::pair<std::set<std::string>::iterator, bool> p;
p = v.insert("ddd");
std::cout << *p.first << " " << std::boolalpha << p.second << std::endl; // ddd true
p = v.insert("aaa");
// p = v.insert(std::move(std::string("aaa")));
std::cout << *p.first << " " << std::boolalpha << p.second << std::endl; // aaa false
  1. 插入 value 到尽可能接近，正好前于 hint 的位置。返回指向被插入元素，或阻止插入的元素的迭代器。

iterator insert( const_iterator hint, const value_type& value );
iterator insert( const_iterator hint, value_type&& value );

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::set<std::string>::iterator it = v.insert(v.begin(), std::string("aab"));
// std::set<std::string>::iterator it = v.insert(v.begin(), std::move(std::string("aba")));
std::cout << *it << std::endl; // aab
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); // aaa aab bbb ccc
  1. 插入来自范围 [first, last) 的元素。 若范围中的多个元素拥有比较等价的关键，则插入哪个元素是未指定的

template< class InputIt > void insert( InputIt first, InputIt last );

std::vector<std::string> v1{"aaa", "bbb", "ccc"};
std::set<std::string> v;
v.insert(v1.begin(), v1.end());
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); // aaa bbb ccc
  1. 插入来自 initializer_list ilist 的元素。 若范围中的多个元素拥有比较等价的关键，则插入哪个元素是未指定的

void insert( std::initializer_list<value_type> ilist );

std::set<std::string> v;
v.insert({"aaa", "bbb", "ccc"});
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); // aaa bbb ccc
  1. 若 nh 是空的结点把柄，则不做任何事。否则插入 nh 所占有的元素到容器，若容器尚未含有拥有等价于 nh.key() 的关键的元素。

insert_return_type insert(node_type&& nh);

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::set<std::string>::node_type nt;
std::set<std::string>::insert_return_type it = v.insert(std::move(nt));
  1. 若 nh 是空的结点把柄，则不做任何事并返回尾迭代器。否则，插入 nh 所占有的元素到容器，若容器尚未含有拥有等价于 nh.key() 的关键的元素，并返回指向拥有等于 nh.key() 的关键的元素的迭代器（无关乎插入成功还是失败）。若插入成功，则从 nh 移动，否则它保持该元素的所有权。元素被插入到尽可能接近正好先于 hint 的位置。

iterator insert(const_iterator hint, node_type&& nh);

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::set<std::string>::node_type nt;
std::set<std::string>::iterator it = v.insert(v.end(), std::move(nt));
std::cout<<*it<<std::endl;

emplace

template< class... Args > std::pair<iterator,bool> emplace( Args&&... args );

若容器中无拥有该关键的元素，则插入以给定的 args 原位构造的新元素到容器。返回由指向被插入元素，或若不发生插入则为既存元素的迭代器，和指代插入是否发生的 bool （若发生插入则为 true ，否则为 false ）。

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::pair<std::set<std::string>::iterator, bool> p;
auto [it, is] = v.insert("ddd");
std::cout << *it << " " << std::boolalpha << is << std::endl; // ddd true
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); // // aaa bbb ccc ddd

emplace_hint

template< class... Args > iterator emplace_hint( const_iterator hint, Args&&... args )

插入新元素到容器中尽可能接近于恰在 hint 前的位置。原位构造元素，即不进行复制或移动操作。返回指向新插入元素的迭代器。若因元素已存在而插入失败，则返回指向拥有等价关键的既存元素的迭代器。

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::pair<std::set<std::string>::iterator, bool> p;
auto it = v.insert(v.begin(), "ddd");
std::cout << *it << std::endl; // ddd
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); // // aaa bbb ccc ddd

erase

从容器擦除指定的元素。

1.移除位于pos的元素。返回指向下一个元素的迭代器。如一处的是最后一个，返回的是指向尾部的迭代器。

iterator erase( iterator pos );

std::set<std::string> v{"aaa", "bbb", "ccc"};
auto it = v.erase(v.begin());
std::cout << *it << std::endl; // bbb
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); // bbb ccc

2.移除一定范围内的元素。返回指向下一个元素的迭代器。如一处的是最后一个，返回的是指向尾部的迭代器.

iterator erase( const_iterator first, const_iterator last );

std::set<std::string> v{"aaa", "bbb", "ccc"};
auto it = v.erase(v.begin(), v.end());
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); //

3.移除关键等于 key 的元素（若存在）。

size_type erase( const key_type& key );

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::set<std::string>::key_type k{"aaa"};
auto s = v.erase(k);
std::cout << s << std::endl; // 1
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); // bbb ccc

swap

void swap( set& other ) noexcept(std::allocator_traits<Allocator>::is_always_equal::value && std::is_nothrow_swappable<Compare>::value)

将内容与其他容器交换。不在单独的元素上调用任何移动、复制或交换操作。所有迭代器和引用保持合法。尾后迭代器被非法化。

std::set<std::string> v1{"ddd", "eee", "fff"};
std::set<std::string> v{"aaa", "bbb", "ccc"};
v.swap(v1);
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); // ddd eee fff

extract

  1. 解链含 position 所指向元素的结点并返回占有它的结点柄。

    node_type extract( const key_type& x );

    std::set<std::string> v{"aaa", "bbb", "ccc"};
std::set<std::string>::key_type k{"aaa"};
std::set<std::string>::node_type nt = v.extract(k);
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); // bbb ccc

  2. 若容器拥有元素而其键等于 x ，则从容器解链该元素并返回占有它的结点句柄。否则，返回空结点句柄。

    node_type extract( const_iterator position );

    std::set<std::string> v{"aaa", "bbb", "ccc"};
std::set<std::string>::node_type nt = v.extract(v.begin());
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; });

merge

template<class C2> void merge( std::set<Key, C2, Allocator>& source );
template<class C2> void merge( std::set<Key, C2, Allocator>&& source );
template<class C2> void merge( std::multiset<Key, C2, Allocator>& source );
template<class C2> void merge( std::multiset<Key, C2, Allocator>&& source );

试图提取其他set中的每个元素，并插入到当前容器 。

std::set<std::string> v1{"ddd", "eee", "fff"};
std::set<std::string> v{"aaa", "bbb", "ccc"};
v.merge(v1);
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; });// aaa bbb ccc ddd eee fff

查找

count

size_type count( const Key& key ) const;
template< class K > size_type count( const K& x ) const，用于查找自定义类型，需重载<比较符。

返回拥有键比较等价于指定参数的元素数，因为此容器不允许重复故为 1 或 0。

std::set<std::string> v{"aaa", "bbb", "ccc"};
std::cout << v.count("aaa") << std::endl; // 1
std::for_each(v.begin(), v.end(), [](std::string a) { std::cout << a << " "; }); // aaa bbb ccc

iterator find( const Key& key );, const_iterator find( const Key& key ) const;
template< class K > iterator find( const K& x );
template< class K > const_iterator find( const K& x ) const;

std::set<std::string> v{"aaa", "bbb", "ccc"};
auto it = v.find("aaa");std::cout << *it << std::endl; // aaa

contains

bool contains( const Key& key ) const;
template< class K > bool contains( const K& x ) const;

检查容器是否含有带特定键的元素。

std::set<std::string> v{"aaa", "bbb", "ccc"};
bool it = v.contains("aaa");

equal_range

std::pair<iterator,iterator> equal_range( const Key& key );
std::pair<const_iterator,const_iterator> equal_range( const Key& key ) const;
template< class K > std::pair<iterator,iterator> equal_range( const K& x );
template< class K > std::pair<const_iterator,const_iterator> equal_range( const K& x ) const;

返回匹配特定键的元素范围.返回值第一个指向首个不小于 key 的元素，第二个指向首个大于key 的元素。

std::set<std::string> v{"aaa", "bbb", "ccc"};
auto [a, b] = v.equal_range("bbb");
std::cout << *a << " " << *b << std::endl; // bbb ccc

lower_bound

iterator lower_bound( const Key& key );, const_iterator lower_bound( const Key& key ) const;
template< class K > iterator lower_bound(const K& x);
template< class K > const_iterator lower_bound(const K& x) const;

返回指向首个不小于(大于、等于)给定键的元素的迭代器.

std::set<std::string> v{"aaa", "bbb", "ccc"};

std::cout << *a << std::endl; // bbb

upper_bound

iterator upper_bound( const Key& key );, const_iterator upper_bound( const Key& key ) const;
