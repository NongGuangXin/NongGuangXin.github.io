C++ string

template< class CharT, class Traits = std::char_traits<CharT>, class Allocator = std::allocator<CharT>
> class basic_string;

namespace pmr {
    template <class CharT, class Traits = std::char_traits<CharT>>
    using basic_string = std::basic_string< CharT, Traits, std::polymorphic_allocator<CharT>>
}

类模板 basic_string 存储并操纵作为非数组平凡标准布局类型的仿 char 对象序列。

成员类型

成员类型 定义
traits_type Traits
value_type CharT
allocator_type Allocator
size_type std::allocator_traits<Allocator>::size_type
difference_type std::allocator_traits<Allocator>::difference_type
reference value_type&
const_reference const value_type&
pointer std::allocator_traits<Allocator>::pointer
const_pointer std::allocator_traits<Allocator>::const_pointer
iterator 遗留随机访问迭代器 (LegacyRandomAccessIterator)
const_iterator 常随机访问迭代器
reverse_iterator std::reverse_iterator<iterator>
const_reverse_iterator std::reverse_iterator<const_iterator>

构造

构造函数

  1. 创建一个空string. 

    constexpr basic_string() noexcept(noexcept( Allocator() )) : basic_string( Allocator() ) {}
explicit constexpr basic_string( const Allocator& alloc ) noexcept;

  2. 创建count个值为ch的容器

    constexpr basic_string( size_type count, CharT ch, const Allocator& alloc = Allocator() );

  3. 以字符串s创建。

    constexpr basic_string( const CharT* s, const Allocator& alloc = Allocator() );

  4. 初始化列表

    constexpr basic_string( std::initializer_list<CharT> ilist,
                        const Allocator& alloc = Allocator() );

  5. 以另一个字符串的子串创建

    constexpr basic_string( const basic_string& other, size_type pos,
                        const Allocator& alloc = Allocator() );

  6. 以一对迭代器构建

    template< class InputIt >
constexpr basic_string( InputIt first, InputIt last, const Allocator& alloc = Allocator() );

  7. 以s的前面count个元素构建

    constexpr basic_string( const CharT* s,size_type count,
                        const Allocator& alloc = Allocator() );

  8. 以其他的vector容器创建

    constexpr basic_string( const basic_string& other );

    还可以使用移动语义

    constexpr basic_string( basic_string&& other ) noexcept;

  9. 其他可隐式转换为string_view的类型

    template < class T >
explicit constexpr basic_string( const T& t, const Allocator& alloc = Allocator() );

template < class T >
constexpr basic_string( const T& t, size_type pos, size_type n,
                        const Allocator& alloc = Allocator() );

get_allocator

返回与容器关联的分配器。

constexpr allocator_type get_allocator() const;

元素访问

at

返回位于指定位置 pos 的元素的引用，有边界检查。若 pos 不在容器范围内，则抛出 std::out_of_range 类型的异常。

constexpr reference       at( size_type pos );
constexpr const_reference at( size_type pos ) const;

operator[]

返回位于指定位置 pos 的元素的引用。不进行边界检查。通过此运算符访问不存在的元素是未定义行为。

constexpr reference       operator[]( size_type pos );
constexpr const_reference operator[]( size_type pos ) const;

front

返回到容器首元素的引用。在空容器上对 front 的调用是未定义的。

constexpr CharT& front();
constexpr const CharT& front() const;

back

返回到容器中最后一个元素的引用。在空容器上调用 back 导致未定义行为。

constexpr CharT& back();
constexpr const CharT& back() const;

data

返回指向作为元素存储工作的底层数组的指针。对于非空容器，返回的指针与首元素地址比较相等。

constexpr const CharT* data() const noexcept;
constexpr CharT* data() noexcept;

c_str

返回字符串的不可修改的 C 字符数组版本，通过 c_str() 写入字符数组是未定义行为。c_str() 与 data() 同一功能。

constexpr const CharT* c_str() const noexcept;

operator basic_string_view

返回到整个字符串的不可修改的 basic_string_view

constexpr operator std::basic_string_view<CharT, Traits>() const noexcept;

迭代器

begin - end

返回指向容器开头和结尾的迭代器。

constexpr iterator begin() noexcept;
constexpr const_iterator begin() const noexcept;

constexpr iterator end() noexcept;
constexpr const_iterator end() const noexcept;

cbegin - cend

返回指向容器开头和结尾的常量迭代器。

constexpr const_iterator cbegin() const noexcept;
constexpr const_iterator cend() const noexcept;

rbegin - rend

返回指向容器开头和结尾的反向迭代器。

constexpr reverse_iterator rbegin() noexcept;
constexpr const_reverse_iterator rbegin() const noexcept;

constexpr reverse_iterator rend() noexcept;
constexpr const_reverse_iterator rend() const noexcept;

crbegin - crend

返回指向容器开头和结尾的反向常量迭代器。

constexpr const_reverse_iterator crbegin() const noexcept;
constexpr const_reverse_iterator crend() const noexcept;

容量

empty

判断容器是否为空，空则为true

s[[nodiscard]] constexpr bool empty() const noexcept;

size

返回容器中的元素数, 即 std::distance(begin(), end()) 。

constexpr size_type size() const noexcept;

length

返回容器中的元素数

constexpr size_type length() const noexcept;

max_size

返回根据系统或库实现限制的容器可保有的元素最大数量

constexpr size_type max_size() const noexcept;

reserve

增加 vector 的容量到大于或等于 new_cap 的值。若 new_cap 大于当前的 capacity() ，则分配新存储.函数无返回值。不可用于减少容量。

constexpr void reserve( size_type new_cap );

capacity

返回容器当前已为之分配空间的元素数。取决于实现。

constexpr size_type capacity() const noexcept;

shrink_to_fit

请求移除未使用的容量。它是减少 capacity() 到 size()非强制性请求。请求是否达成依赖于实现。

constexpr void shrink_to_fit();

操作

clear

擦除所有元素。此调用后 size() 返回零。非法化任何指代所含元素的引用、指针或迭代器。任何尾后迭代器亦被非法化

constexpr void clear() noexcept;

insert

插入元素到容器中的指定位置。

1.在index前插入s

constexpr basic_string& insert( size_type index, const CharT* s )

2.在 pos 前插入 ch 的 count 个副本。

constexpr basic_string& insert( size_type index, size_type count, CharT ch );

3.在 index插入[s, s+count] .

constexpr basic_string& insert( size_type index, const CharT* s, size_type count );

4.在 index处插入str. 

constexpr basic_string& insert( size_type index, const basic_string& str );

5.在index处插入str[index_str, index_str+count]

constexpr basic_string& insert( size_type index, const basic_string& str,
                                size_type index_str, size_type count = npos);

6.在pos前插入ch

constexpr iterator insert( const_iterator pos, CharT ch );

7.在pos前count个ch.

constexpr iterator insert( const_iterator pos, size_type count, CharT ch );

8.在pos前插入[first, last]

template< class InputIt >
constexpr iterator insert( const_iterator pos, InputIt first, InputIt last );

9.在pos前插入ilist.

constexpr iterator insert( const_iterator pos, std::initializer_list<CharT> ilist );

10.在pos前插入t

template < class T >
constexpr basic_string& insert( size_type pos, const T& t );

11.在pos前插入t[index_str, index+count].

template < class T >
constexpr basic_string& insert( size_type index, const T& t,
                                size_type index_str, size_type count = npos);

erase

从容器擦除指定的元素。

1.移除[index, index+count].

constexpr basic_string& erase( size_type index = 0, size_type count = npos );

2.移除位于pos元素。

constexpr iterator erase( const_iterator position );

3.移除[first, last].

constexpr iterator erase( const_iterator first, const_iterator last );

push_back

添加元素到容器尾。

constexpr void push_back( CharT ch );

pop_back

移除容器的末元素。在空容器上调用 pop_back 导致未定义行为。非法化指向末元素的迭代器和引用，以及 end() 迭代器。

constexpr void pop_back();

resize

重设容器大小以容纳 count 个元素.若当前元素大于 count ，则减小容器为其首 count 个元素。若当前容量小于count则插入默认 元素或指定的ch值。

constexpr void resize( size_type count );
constexpr void resize( size_type count, CharT ch );

swap

将内容与其他容器交换。可能非法化所有迭代器和引用。

constexpr void swap( basic_string& other ) noexcept();

常量
